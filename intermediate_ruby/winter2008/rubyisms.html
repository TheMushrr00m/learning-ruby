  210 #don't reinvent the wheel, check the built in api
  211 def stringify_array(an_array)
  212   a_string = ""
  213   for element in an_array
  214     if a_string.size > 0
  215       a_string += ", "
  216     end
  217     a_string += element.to_s
  218   end
  219   return a_string
  220 end
  221
